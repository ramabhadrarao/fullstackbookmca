üìö What is GraphQL?

GraphQL is a query language for APIs that allows clients to request exactly the data they need.

‚úÖ Advantages

  • Request exact data needed
  • Single endpoint for all operations
  • Strongly typed schema
  • No over-fetching or under-fetching
  • Real-time updates with subscriptions

üìå Key Concepts

  • Query: Read data
  • Mutation: Create/Update/Delete data
  • Schema: Define data structure
  • Resolver: Functions to fetch data

üìã GraphQL Schema Definition

File: server/schema/schema.js

const graphql = require('graphql');
const User = require('../models/User');

const {
  GraphQLObjectType,
  GraphQLString,
  GraphQLInt,
  GraphQLID,
  GraphQLList,
  GraphQLSchema,
  GraphQLNonNull
} = graphql;

// ========================================
// USER TYPE DEFINITION
// ========================================
const UserType = new GraphQLObjectType({
  name: 'User',
  fields: () => ({
    id: { type: GraphQLID },
    name: { type: GraphQLString },
    email: { type: GraphQLString },
    age: { type: GraphQLInt },
    city: { type: GraphQLString }
  })
});

// ========================================
// ROOT QUERY (READ OPERATIONS)
// ========================================
const RootQuery = new GraphQLObjectType({
  name: 'RootQueryType',
  fields: {
    
    // Get Single User by ID
    user: {
      type: UserType,
      args: { id: { type: GraphQLID } },
      resolve(parent, args) {
        return User.findById(args.id);
      }
    },
    
    // Get All Users
    users: {
      type: new GraphQLList(UserType),
      resolve(parent, args) {
        return User.find({});
      }
    },
    
    // Get Users by City
    usersByCity: {
      type: new GraphQLList(UserType),
      args: { city: { type: GraphQLString } },
      resolve(parent, args) {
        return User.find({ city: args.city });
      }
    }
  }
});

// ========================================
// MUTATIONS (CREATE, UPDATE, DELETE)
// ========================================
const Mutation = new GraphQLObjectType({
  name: 'Mutation',
  fields: {
    
    // Add New User
    addUser: {
      type: UserType,
      args: {
        name: { type: new GraphQLNonNull(GraphQLString) },
        email: { type: new GraphQLNonNull(GraphQLString) },
        age: { type: new GraphQLNonNull(GraphQLInt) },
        city: { type: new GraphQLNonNull(GraphQLString) }
      },
      resolve(parent, args) {
        const user = new User({
          name: args.name,
          email: args.email,
          age: args.age,
          city: args.city
        });
        return user.save();
      }
    },
    
    // Update User
    updateUser: {
      type: UserType,
      args: {
        id: { type: new GraphQLNonNull(GraphQLID) },
        name: { type: GraphQLString },
        email: { type: GraphQLString },
        age: { type: GraphQLInt },
        city: { type: GraphQLString }
      },
      resolve(parent, args) {
        return User.findByIdAndUpdate(
          args.id,
          {
            $set: {
              name: args.name,
              email: args.email,
              age: args.age,
              city: args.city
            }
          },
          { new: true }
        );
      }
    },
    
    // Delete User
    deleteUser: {
      type: UserType,
      args: {
        id: { type: new GraphQLNonNull(GraphQLID) }
      },
      resolve(parent, args) {
        return User.findByIdAndDelete(args.id);
      }
    }
  }
});

// ========================================
// EXPORT SCHEMA
// ========================================
module.exports = new GraphQLSchema({
  query: RootQuery,
  mutation: Mutation
});

const express = require('express');
const { graphqlHTTP } = require('express-graphql');
const schema = require('./schema/schema');
const mongoose = require('mongoose');
const cors = require('cors');

const app = express();
const PORT = 5000;

// Enable CORS
app.use(cors());

// MongoDB Connection

  useNewUrlParser: true,
  useUnifiedTopology: true
});

mongoose.connection.once('open', () => {
  console.log('‚úÖ Connected to MongoDB');
});

// GraphQL Endpoint
app.use('/graphql', graphqlHTTP({
  schema: schema,
  graphiql: true  // Enable GraphiQL interface
}));

// Start Server
app.listen(PORT, () => {
  console.log(`üöÄ Server running on http://localhost:${PORT}`);
  console.log(`üìä GraphiQL interface: http://localhost:${PORT}/graphql`);
});

üì± React Component - Display Users

File: client/src/components/UserList.js

import React from 'react';
import { useQuery, gql } from '@apollo/client';

// Define GraphQL Query
const GET_USERS = gql`
  query GetUsers {
    users {
      id
      name
      email
      age
      city
    }
  }
`;

function UserList() {
  const { loading, error, data } = useQuery(GET_USERS);

  if (loading) return <p>Loading...</p>;
  if (error) return <p>Error: {error.message}</p>;

  return (
    <div>
      <h2>User List</h2>
      <table className="table">
        <thead>
          <tr>
            <th>Name</th>
            <th>Email</th>
            <th>Age</th>
            <th>City</th>
          </tr>
        </thead>
        <tbody>
          {data.users.map(user => (
            <tr key={user.id}>
              <td>{user.name}</td>
              <td>{user.email}</td>
              <td>{user.age}</td>
              <td>{user.city}</td>
            </tr>
          ))}
        </tbody>
      </table>
    </div>
  );
}

export default UserList;

‚ûï React Component - Add User

File: client/src/components/AddUser.js

import React, { useState } from 'react';
import { useMutation, gql } from '@apollo/client';

// Define GraphQL Mutation
const ADD_USER = gql`
  mutation AddUser($name: String!, $email: String!, $age: Int!, $city: String!) {
    addUser(name: $name, email: $email, age: $age, city: $city) {
      id
      name
      email
      age
      city
    }
  }
`;

// Refetch query after mutation
const GET_USERS = gql`
  query GetUsers {
    users {
      id
      name
      email
      age
      city
    }
  }
`;

function AddUser() {
  const [formData, setFormData] = useState({
    name: '',
    email: '',
    age: '',
    city: ''
  });

  const [addUser, { loading, error }] = useMutation(ADD_USER, {
    refetchQueries: [{ query: GET_USERS }]
  });

  const handleChange = (e) => {
    setFormData({
      ...formData,
      [e.target.name]: e.target.value
    });
  };

  const handleSubmit = (e) => {
    e.preventDefault();
    addUser({
      variables: {
        name: formData.name,
        email: formData.email,
        age: parseInt(formData.age),
        city: formData.city
      }
    });
    // Reset form
    setFormData({ name: '', email: '', age: '', city: '' });
  };

  return (
    <div>
      <h2>Add New User</h2>
      <form onSubmit={handleSubmit}>
        <div className="mb-3">
          <label>Name:</label>
          <input 
            type="text" 
            name="name" 
            value={formData.name}
            onChange={handleChange}
            className="form-control"
            required
          />
        </div>
        
        <div className="mb-3">
          <label>Email:</label>
          <input 
            type="email" 
            name="email" 
            value={formData.email}
            onChange={handleChange}
            className="form-control"
            required
          />
        </div>
        
        <div className="mb-3">
          <label>Age:</label>
          <input 
            type="number" 
            name="age" 
            value={formData.age}
            onChange={handleChange}
            className="form-control"
            required
          />
        </div>
        
        <div className="mb-3">
          <label>City:</label>
          <input 
            type="text" 
            name="city" 
            value={formData.city}
            onChange={handleChange}
            className="form-control"
            required
          />
        </div>
        
        <button type="submit" className="btn btn-primary" disabled={loading}>
          {loading ? 'Adding...' : 'Add User'}
        </button>
      </form>
      {error && <p className="text-danger mt-2">Error: {error.message}</p>}
    </div>
  );
}

export default AddUser;

Feature REST API GraphQL
Endpoints Multiple endpoints
(/users, /posts, /comments)		 Single endpoint
(/graphql)
Data Fetching Fixed data structure

‚ú® Best Practices

‚úÖ DO

  • Use descriptive names for queries and mutations
  • Validate input data
  • Handle errors gracefully
  • Use fragments for reusable fields
  • Implement pagination for large datasets
  • Add authentication and authorization
  • Use DataLoader to prevent N+1 queries

‚ùå DON'T

  • Don't fetch all data at once (use pagination)
  • Don't expose sensitive information
  • Don't create deeply nested queries
  • Don't ignore error handling
  • Don't skip input validation
  • Don't use GraphQL for file uploads (use REST)

üö® Error Handling

Server-Side Error Handling

resolve(parent, args) {
  try {
    // Check if user exists
    const existingUser = await User.findOne({ email: args.email });
    if (existingUser) {
      throw new Error('Email already exists');
    }
    
    // Create user
    const user = new User({ ...args });
    return await user.save();
    
  } catch (error) {
    throw new Error(`Failed to add user: ${error.message}`);
  }
}

Client-Side Error Handling

const [addUser, { loading, error }] = useMutation(ADD_USER, {
  onCompleted: (data) => {
    console.log('User added successfully:', data);
    alert('User added!');
  },
  onError: (error) => {
    console.error('Error adding user:', error);
    alert(`Error: ${error.message}`);
  }
});
